#include <bits/stdc++.h>
using namespace std;
#define task "MESSAGE"
#define ll long long int
#define pb push_back
#define el "\n"
#define vpll vector<pair<ll, ll>> 
#define vll vector<long long>
#define pii pair<int, int>
#define pll pair<ll, ll>
#define sll stack<ll>
#define qll queue<ll>
#define dqll deque<ll>
#define pq priority_queue
const ll N = 1e6 + 2;
const ll M = 1e3 + 2;
const ll MOD = 2111992;
ll n, x, y;
vll adj[M];
vll roots;
bool visited[N];
int maxs = 0;
void tassk() {
    ios_base::sync_with_stdio(0);
    cin.tie(0); cout.tie(0);
    if (fopen(task".inp", "r")) {
        freopen(task".inp", "r", stdin);
        freopen(task".out", "w", stdout);
    }
}
void dfs(ll u){
    visited[u] = 1;
    for (auto v : adj[u]){
        if (!visited[v]){
            dfs(v);
        }
    }
}
void connected(){
    ll dem = 0;
    for (int i = 1; i <= n; i++){
        if (!visited[i]){
            dem++;
            roots.pb(i);
            dfs(i);
        }
    }
    cout << dem << el;
    for (auto i : roots) cout << i << " ";
}
void solve() {
    cin >> n;
    for (int i = 1; i <= n; i++){
        cin >> x >> y;
        adj[x].pb(y);
        adj[y].pb(x);
    }
    connected();
}

int main() {
    tassk();
    solve();
    return 0;
}
