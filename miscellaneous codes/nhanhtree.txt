#include <bits/stdc++.h>
using namespace std;
const int MAX = 1e5+5;
int segMin[4*MAX],segMax[4*MAX],segSum[4*MAX],a[MAX],n,q;
void udmin(int id,int l,int r){
    if(l == r) {
        segMin[id] = a[l];
        return;
    }
    int mid = (l + r) / 2;
    udmin(2 * id, l, mid);
    udmin(2 * id + 1, mid + 1, r);
    segMin[id] = min(segMin[2 * id], segMin[2 * id + 1]);
}
void udmax(int id, int l, int r){
    if (l == r) {
        segMax[id] = a[l];
        return;
    }
    int mid = (l + r) / 2;
    udmax(2 * id, l, mid);
    udmax(2 * id + 1, mid + 1, r);
    segMax[id] = max(segMax[2 * id], segMax[2 * id + 1]);
}
void udsum(int id, int l, int r) {
    if (l == r) {
        segSum[id] = a[l];
        return;
    }
    int mid = (l + r) / 2;
    udsum(2 * id, l, mid);
    udsum(2 * id + 1, mid + 1, r);
    segSum[id] = segSum[2 * id] + segSum[2 * id + 1];
}
void update(int id, int l, int r, int i, int val) {
    if (l == r) {
        segMin[id] = segMax[id] = segSum[id] = val;
        return;
    }
    int mid = (l + r) / 2;
    if (i <= mid) update(2 * id, l, mid, i, val);
    else update(2 * id + 1, mid + 1, r, i, val);
    segMin[id] = min(segMin[2 * id], segMin[2 * id + 1]);
    segMax[id] = max(segMax[2 * id], segMax[2 * id + 1]);
    segSum[id] = segSum[2 * id] + segSum[2 * id + 1];
}
int qumin(int id, int l, int r, int u, int v) {
    if (v < l || u > r) return INT_MAX;
    if (u <= l && r <= v) return segMin[id];
    int mid = (l + r) / 2;
    return min(qumin(2 * id, l, mid, u, v), qumin(2 * id + 1, mid + 1, r, u, v));
}
int qumax(int id, int l, int r, int u, int v) {
    if (v < l || u > r) return INT_MIN;
    if (u <= l && r <= v) return segMax[id];
    int mid = (l + r) / 2;
    return max(qumax(2 * id, l, mid, u, v), qumax(2 * id + 1, mid + 1, r, u, v));
}
int qusum(int id, int l, int r, int u, int v) {
    if (v < l || u > r) return 0;
    if (u <= l && r <= v) return segSum[id];
    int mid = (l + r) / 2;
    return qusum(2 * id, l, mid, u, v) + qusum(2 * id + 1, mid + 1, r, u, v);
}

int main() {
    cin >> n >> q;
    for (int i = 1; i <= n; ++i) {
        cin >> a[i];
    }
    udmin(1, 1, n);
    udmax(1, 1, n);
    udsum(1, 1, n);
    while (q--) {
        char type;
        int l, r;
        cin >> type >> l >> r;
        if (type == 'C') {
            update(1, 1, n, l, r);
        } else if (type == 'L') {
            cout << qumin(1, 1, n, l, r) << endl;
        } else if (type == 'H') {
            cout << qumax(1, 1, n, l, r) << endl;
        } else if (type == 'S') {
            cout << qusum(1, 1, n, l, r) << endl;
        }
    }

    return 0;
}
